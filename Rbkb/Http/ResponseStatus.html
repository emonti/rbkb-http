def capture(str)
  raise "arg 0 must be a string" unless str.is_a?(String)
  unless m=/^([^\s]+)\s+(\d+)(?:\s+(.*))?$/.match(str)
    raise "invalid status #{str.inspect}"
  end
  @version = m[1]
  @code = m[2] =~ /^\d+$/ ? m[2].to_i : m[2]
  @text = m[3]
  return self
end
