def capture(str)
  raise "arg 0 must be a string" unless str.is_a? String
  str.split(/\s*;\s*/).each do |p|
    var, val = p.split('=', 2)
    if val =~ /^(['"])(.*)\1$/
      val = $2
    end
    self << [var.strip, val]
  end
  return self
end

def to_raw(quote_val=false)
  ret = ([nil] + self).map do |k,v| 
    if v 
      "#{k}=#{quote_val ? "\"#{v}\"" : v}"
    else 
      "#{k}" 
    end
  end.join("; ")
end
