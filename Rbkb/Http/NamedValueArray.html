def [](*args)
  if args.size == 1 and (String === args[0] or Symbol === args[0])
    if h=find {|x| x[0] == args[0]}
      return h[1]
    end
  else
    super(*args)
  end
end

def []=(*args)
  if args.size > 1 and (String === args[0] or Symbol === args[0])
    if h=find {|x| x[0] == args[0]}
      h[1] = args[1]
    else
      self << args[0,2]
    end
  else
    super(*args)
  end
end
