def capture(str)
  raise "arg 0 must be a string" unless str.is_a?(String)
  unless m=/^([^\s]+)\s+([^\s]+)(?:\s+([^\s]+))?\s*$/.match(str)
    raise "invalid action #{str.inspect}"
  end
  @verb = m[1]
  @uri = URI.parse m[2]
  @version = m[3]
  return self
end

def to_raw

  ary << @version if @version
  ary.join(" ")
end
