def capture(str, url_dec=false)
  raise "arg 0 must be a string" unless str.is_a? String
  str.split('&').each do |p| 
    k,v = p.split('=',2)
    if url_dec
      k = k.urldec
      v = v.urldec
    end
    self << [k, v]
  end
  return self
end

def to_raw(url_enc=false)
  self.map do |k,v|
    if url_enc
      k = k.urlenc
      v = v.urlenc
    end
    if v
      "#{k}=#{v}" 
    else 
      "#{k}"
    end
  end.join('&')
end
